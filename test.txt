-- Wait for the game to load
wait()

-- Get the CurrentFloor model in the Workspace
local currentFloor = workspace:FindFirstChild("CurrentFloor")

if currentFloor then
    -- Find the first model in CurrentFloor
    local firstModel = currentFloor:FindFirstChildWhichIsA("Model")
    
    if firstModel then
        -- Find the folder named "monsters" within the first model
        local monstersFolder = firstFolder:FindFirstChild("monsters")
        
        if monstersFolder then
            -- Create a ScreenGui for the GUI
            local screenGui = Instance.new("ScreenGui")
            screenGui.Parent = game.Players.LocalPlayer:WaitForChild("PlayerGui")
            
            -- Iterate through the models inside the monsters folder
            for _, monster in pairs(monstersFolder:GetChildren()) do
                if monster:IsA("Model") then
                    -- Create a TextButton for each monster
                    local button = Instance.new("TextButton")
                    button.Size = UDim2.new(0, 200, 0, 50)
                    button.Position = UDim2.new(1, -210, 0, 50 * _)
                    button.Text = monster.Name
                    button.Parent = screenGui
                    
                    -- Connect the button click event
                    button.MouseButton1Click:Connect(function()
                        -- Set the monster's position to (0, 0, 0)
                        if monster:FindFirstChild("PrimaryPart") then
                            monster:SetPrimaryPartCFrame(CFrame.new(0, 0, 0))
                        else
                            -- If the model doesn't have a PrimaryPart, move all parts to (0, 0, 0)
                            for _, part in pairs(monster:GetDescendants()) do
                                if part:IsA("BasePart") then
                                    part.CFrame = CFrame.new(0, 0, 0)
                                end
                            end
                        end
                    end)
                end
            end
        else
            warn("Monsters folder not found in the first model of CurrentFloor.")
        end
    else
        warn("No models found in CurrentFloor.")
    end
else
    warn("CurrentFloor not found in the Workspace.")
end
